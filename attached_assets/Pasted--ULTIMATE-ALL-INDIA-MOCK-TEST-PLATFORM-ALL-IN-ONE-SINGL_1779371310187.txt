<script>

/* ====================================
QUESTION DATABASE
ADD UNLIMITED QUESTIONS
==================================== */

const allQuestions = [];

/* ====================================
AUTO GENERATE 100 QUESTIONS PER EXAM
==================================== */

const exams = [

"JEE Main",
"NEET UG",
"UPSC",
"SSC CGL",
"Bank PO",
"Railway",
"OSSSC",
"OPSC",
"UPPSC",
"BPSC",
"MPSC",
"TNPSC",
"Police Exam"

];

exams.forEach(exam=>{

for(let i=1;i<=100;i++){

allQuestions.push({

exam:exam,

question:{

English:`${exam} Question ${i}`,
Hindi:`${exam} प्रश्न ${i}`,
Odia:`${exam} ପ୍ରଶ୍ନ ${i}`

},

options:{

English:[
"Option A",
"Option B",
"Option C",
"Option D"
],

Hindi:[
"विकल्प A",
"विकल्प B",
"विकल्प C",
"विकल्प D"
],

Odia:[
"ବିକଳ୍ପ A",
"ବିକଳ୍ପ B",
"ବିକଳ୍ପ C",
"ବିକଳ୍ପ D"
]

},

answer:Math.floor(Math.random()*4)

});

}

});

/* ====================================
QUIZ VARIABLES
==================================== */

let questions = [];
let currentQuestion = 0;
let score = 0;
let selected = null;
let answers = [];
let language = "English";
let timer;
let timeLeft = 300;

/* ====================================
START TEST
==================================== */

function startTest(){

language =
document.getElementById("language").value;

const exam =
document.getElementById("exam").value;

const count =
parseInt(
document.getElementById("questionCount").value
);

questions =
allQuestions.filter(
q=>q.exam===exam
);

questions =
shuffleArray(questions);

questions =
questions.slice(0,count);

document.getElementById("setup")
.classList.add("hidden");

document.getElementById("quizBox")
.classList.remove("hidden");

document.getElementById("totalQ")
.innerText = questions.length;

timeLeft =
parseInt(
document.getElementById("time").value
)*60;

startTimer();

loadQuestion();

}

/* ====================================
SHUFFLE
==================================== */

function shuffleArray(array){

for(let i=array.length-1;i>0;i--){

const j =
Math.floor(Math.random()*(i+1));

[array[i],array[j]] =
[array[j],array[i]];

}

return array;

}

/* ====================================
TIMER
==================================== */

function startTimer(){

updateTimer();

timer = setInterval(()=>{

timeLeft--;

updateTimer();

if(timeLeft<=0){

finishTest();

}

},1000);

}

function updateTimer(){

const m =
Math.floor(timeLeft/60);

const s =
timeLeft%60;

document.getElementById("timer")
.innerText =
`${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;

}

/* ====================================
LOAD QUESTION
==================================== */

function loadQuestion(){

selected = null;

const q =
questions[currentQuestion];

document.getElementById("currentQ")
.innerText =
currentQuestion+1;

document.getElementById("question")
.innerText =
q.question[language];

const optionsBox =
document.getElementById("options");

optionsBox.innerHTML = "";

q.options[language]
.forEach((option,index)=>{

const div =
document.createElement("div");

div.classList.add("option");

div.innerText = option;

div.onclick = ()=>{

document.querySelectorAll(".option")
.forEach(o=>
o.classList.remove("selected")
);

div.classList.add("selected");

selected = index;

};

optionsBox.appendChild(div);

});

}

/* ====================================
NEXT QUESTION
==================================== */

function nextQuestion(){

if(selected===null){

alert("Please select option");

return;

}

answers.push(selected);

if(
selected===
questions[currentQuestion].answer
){

score++;

}

currentQuestion++;

if(
currentQuestion>=questions.length
){

finishTest();

}else{

loadQuestion();

}

}

/* ====================================
FINISH TEST
==================================== */

function finishTest(){

clearInterval(timer);

document.getElementById("quizBox")
.classList.add("hidden");

document.getElementById("resultBox")
.classList.remove("hidden");

document.getElementById("finalScore")
.innerText =
`${score} / ${questions.length}`;

let html =
"<h2>📘 Answer Review</h2>";

questions.forEach((q,index)=>{

html += `

<div class="answer-item">

<b>Q${index+1}:</b>

${q.question[language]}

<br><br>

✅ Correct Answer:
<b>
${q.options[language][q.answer]}
</b>

<br><br>

🧠 Your Answer:
<b>
${q.options[language][answers[index]]}
</b>

</div>

`;

});

document.getElementById("answerReview")
.innerHTML = html;

}
